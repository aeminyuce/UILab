#### Installed packages:
```
// Less
npm i -D less less-plugin-clean-css

// Javascript
npm i -D @babel/core @babel/cli @babel/node @babel/preset-env
npm i -D eslint @babel/eslint-parser

// Fonts
npm i @fontsource/poppins @fontsource/archivo
```
